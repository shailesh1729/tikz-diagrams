          }

      % Dictionary
      \begin{scope}[scale=\sc]
      \foreach \r in {0,...,\Na}{
          \foreach \c in {0,...,\Da}{
              \pgfmathparse{0.8*rnd+0.1};
              \definecolor{MyColor}{rgb}{\pgfmathresult,\pgfmathresult,\pgfmathresult}
